def sus(array):
    respuestas = json.loads(array)
    suma = 0
    lista = respuestas+[['','']]
    for x in range(len(lista)):
        if (x+1)%2==0:
            suma+=5-resp_usabilidad[lista[x][1]]
        else:
            suma+=resp_usabilidad[lista[x][1]]-1
    sus = suma * 2.5
    return sus

def graficar_nps(df,medico=True):
    category_names = ['Detractores','Pasivos','Promotores']
    results = {
        'NPS':[len(df[df['NPS']<=6]),len(df[(df['NPS']>=7)&(df['NPS']<9)]),len(df[df['NPS']>=9])]
    }


    def survey(results, category_names):
        labels = list(results.keys())
        data = np.array(list(results.values()))
        data_cum = data.cumsum(axis=1)
        category_colors = plt.get_cmap('RdYlGn')(
            np.linspace(0.15, 0.85, data.shape[1]))

        fig, ax = plt.subplots(figsize=(9.2, 5))
        ax.invert_yaxis()
        ax.xaxis.set_visible(False)
        ax.set_xlim(0, np.sum(data, axis=1).max())

        for i, (colname, color) in enumerate(zip(category_names, category_colors)):
            widths = data[:, i]
            starts = data_cum[:, i] - widths
            rects = ax.barh(labels, widths, left=starts, height=0.5,
                            label=colname, color=color)

            r, g, b, _ = color
            text_color = 'white' if r * g * b < 0.5 else 'darkgrey'
            ax.bar_label(rects, label_type='center', color=text_color)
        ax.legend(ncol=len(category_names), bbox_to_anchor=(0, 1),
                  loc='lower left', fontsize='small')
        fig.set_size_inches(10.5, 1)
        promotores = len(df[df['NPS']>=9])/len(df)
        detractores = len(df[df['NPS']<=6])/len(df)
        
        txt = 'Médicos' if medico else 'Pacientes'
        
        fig.suptitle(f'NPS {txt} = Promotores% - Detractores% = {round((promotores-detractores)*100,2)}%', size=16, y=1.5)
        return fig, ax


    survey(results, category_names)
    plt.show()
