from nltk.sentiment.vader import SentimentIntensityAnalyzer
import nltk
from google_trans_new import google_translator

def sentimientos(txt):
    try:
        translator = google_translator()  
        translate_text = translator.translate(txt, lang_src='es', lang_tgt='en')
        senti = SentimentIntensityAnalyzer()
        kvp = senti.polarity_scores(translate_text)
        max_l = ""
        max_n = 0
        for x in kvp:
            if x!='compound' and x!='neu':
                if kvp[x]>max_n:
                    max_n = kvp[x]
                    max_l = x
        if max_l == "neg":
            return 0
        elif max_l == 'pos':
            return 1
        else:
            return 0.5
#         return (kvp['compound']+1)/2
    except Exception as e:
#         print(e)
        return 0

def procesar_encuesta(encuesta):
    encuesta['Por favor, justifique su respuesta anterior.'] = encuesta['Por favor, justifique su respuesta anterior.'].apply(sentimientos)
    encuesta['Comentarios generales'] = encuesta['Comentarios generales'].apply(sentimientos)
    encuesta[encuesta.columns[6]]=normalizar_col(encuesta[encuesta.columns[6]])
    encuesta[encuesta.columns[8]]=normalizar_col(encuesta[encuesta.columns[8]])
    encuesta[encuesta.columns[9]]=normalizar_col(encuesta[encuesta.columns[9]])
    encuesta[encuesta.columns[11]]=normalizar_col(encuesta[encuesta.columns[11]])
    cols = list(encuesta.columns[[5,6,7,8,9,10,11]])+['Comentarios generales']
    encuesta['ACSI'] = (encuesta[cols].mean(axis=1)*100).apply(int)
    return encuesta[encuesta.columns[[1,2,3,4,-1]]]

def proc_grup(x):
    if 0.9 <= x:
        return "Mayor al 90%"
    elif 0.5<x<0.9:
        return "Entre 50% y 90%"
    else:
        return "Menor al 50%"

apoyo = pacientes_2[pacientes_2.columns[[-6,-5,-4,-3]]]
